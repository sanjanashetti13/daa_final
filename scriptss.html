// Priority Queue Example for High Water Consumption Areas
void identifyHighUsageAreas(vector usageData) {
    priority_queue pq;
    for (int usage : usageData) pq.push(usage);

    cout << "Top 3 High Water Consumption Areas:" << endl;
    for (int i = 0; i < 3 && !pq.empty(); ++i) {
        cout << "Area " << i + 1 << ": " << pq.top() << " liters" << endl;
        pq.pop();
    }
}

// Segment Tree Example for Crop Water Planning
class SegmentTree {
    vector tree;
    int n;

public:
    SegmentTree(vector& data) {
        n = data.size();
        tree.resize(4 * n);
        build(data, 0, n - 1, 0);
    }

    void build(vector& data, int start, int end, int node) {
        if (start == end) {
            tree[node] = data[start];
            return;
        }
        int mid = (start + end) / 2;
        build(data, start, mid, 2 * node + 1);
        build(data, mid + 1, end, 2 * node + 2);
        tree[node] = tree[2 * node + 1] + tree[2 * node + 2];
    }

    int query(int l, int r, int start, int end, int node) {
        if (l > end || r < start) return 0;
        if (l <= start && r >= end) return tree[node];

        int mid = (start + end) / 2;
        int leftSum = query(l, r, start, mid, 2 * node + 1);
        int rightSum = query(l, r, mid + 1, end, 2 * node + 2);
        return leftSum + rightSum;
    }

    int query(int l, int r) {
        return query(l, r, 0, n - 1, 0);
    }
};

int main() {
    vector usageData = {500, 800, 1200, 300, 600};
    identifyHighUsageAreas(usageData);

    vector cropWaterRequirements = {200, 150, 400, 300, 250};
    SegmentTree st(cropWaterRequirements);
    cout << "Total Water Required for Crops [Index 1-3]: " << st.query(1, 3) << " liters" << endl;

    return 0;
}
