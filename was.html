Implemented Code


#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

void identifyHighConsumption(vector<pair<int, string>> &areas) {
    // Sort areas by water consumption in descending order
    sort(areas.begin(), areas.end(), greater<>());

    cout << "Areas with high water consumption:<br>" << endl;
    for (auto &area : areas) {
        cout << "Area: " << area.second << ", Consumption: " << area.first << " liters<br>" << endl;
    }
}

int main() {
    // Example data: (water consumption, area name)
    vector<pair<int, string>> areas = {
        {3000, "Area A"},
        {4500, "Area B"},
        {2000, "Area C"},
        {5000, "Area D"}
    };

    identifyHighConsumption(areas);

    return 0;
}
