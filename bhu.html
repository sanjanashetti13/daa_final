Jaipur Rainfall Data Management

A robust system to efficiently record, store, and analyze daily rainfall data for Jaipur using advanced data structures.

Objectives for Jaipur Rainfall Data

  • Record daily rainfall data for Jaipur with efficient storage and retrieval mechanisms.
  • Analyze rainfall trends and provide insights for future urban planning in Jaipur.
  • Use data structures to optimize performance and scalability specific to Jaipur's needs.

Implementation Highlights for Jaipur Rainfall Management

Our implementation utilizes advanced data structures such as Arrays, Linked Lists, Hash Tables, and Binary Search Trees (BSTs) to manage Jaipur's rainfall data efficiently. These tools ensure real-time data access and efficient processing tailored to the climate patterns of Jaipur.

// 1. Array Implementation for storing rainfall data
void analyzeRainfallUsingArray(const vector& data) {
    int totalRainfall = 0;
    for (int month : data) totalRainfall += month;
    cout << "Total Rainfall (Array): " << totalRainfall << " mm" << endl;
}

// 2. Hash Table for month-wise rainfall lookup
void analyzeRainfallUsingHashTable(const vector& data, const vector& months) {
    unordered_map rainfallData;
    for (size_t i = 0; i < data.size(); ++i)
        rainfallData[months[i]] = data[i];

    cout << "Rainfall Data (Hash Table):" << endl;
    for (const auto& entry : rainfallData)
        cout << entry.first << ": " << entry.second << " mm" << endl;
}

// 3. Binary Search Tree (BST) for rainfall data storage
struct TreeNode {
    int data;
    TreeNode* left = nullptr;
    TreeNode* right = nullptr;
    TreeNode(int val) : data(val) {}
};

class BST {
public:
    TreeNode* root = nullptr;

    void insert(int data) {
        root = insertNode(root, data);
    }

    void inOrderTraversal(TreeNode* node) {
        if (node) {
            inOrderTraversal(node->left);
            cout << node->data << " ";
            inOrderTraversal(node->right);
        }
    }

private:
    TreeNode* insertNode(TreeNode* node, int data) {
        if (!node) return new TreeNode(data);
        if (data < node->data) node->left = insertNode(node->left, data);
        else node->right = insertNode(node->right, data);
        return node;
    }
};

int main() {
    vector rainfallData = {22, 18, 30, 25, 35, 100, 200, 220, 150, 40};
    vector months = {"Jan", "Feb", "Mar", "Apr", "May", "Jun", "Jul", "Aug", "Sep", "Oct"};

    analyzeRainfallUsingArray(rainfallData);
    analyzeRainfallUsingHashTable(rainfallData, months);

    BST bst;
    for (int data : rainfallData) bst.insert(data);
    cout << "Rainfall Data (BST - In-Order): ";
    bst.inOrderTraversal(bst.root);
    cout << endl;

    return 0;
}
