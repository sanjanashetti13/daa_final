Jaipur Water Distribution Management

A comprehensive solution for efficient water distribution across Jaipur using advanced data structures and algorithms.

Efficient Water Distribution Objectives

  • Ensure fair and efficient water distribution across Jaipur neighborhoods.
  • Implement MST to minimize pipeline installation costs.
  • Use Shortest Path Tree for optimal water flow routes.

Implemented Code

#include <iostream>
#include <vector>
#include <queue>
#include <climits>
using namespace std;

const int INF = INT_MAX;

void primMST(vector>> &graph, int n) {
    vector key(n, INF);
    vector inMST(n, false);
    key[0] = 0;

    priority_queue, vector>, greater<>> pq;
    pq.push({0, 0});

    while (!pq.empty()) {
        int u = pq.top().second;
        pq.pop();

        if (inMST[u]) continue;
        inMST[u] = true;

        for (auto &[v, weight] : graph[u]) {
            if (!inMST[v] && weight < key[v]) {
                key[v] = weight;
                pq.push({key[v], v});
            }
        }
    }
}
int main() {
    vector>> graph(5);
    graph[0].push_back({1, 2}); graph[1].push_back({0, 2});
    primMST(graph);
}
